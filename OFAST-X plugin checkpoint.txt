**Database Table:**
```sql
CREATE TABLE wp_ofast_snippets (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    code LONGTEXT NOT NULL,
    language ENUM('php', 'javascript', 'css') DEFAULT 'php',
    active TINYINT(1) DEFAULT 0,
    tags TEXT,
    scope ENUM('global', 'admin', 'frontend') DEFAULT 'global',
    created_at DATETIME,
    updated_at DATETIME,
    INDEX idx_active (active),
    INDEX idx_language (language)
```

## 🔐 LICENSE ENFORCEMENT (STRICT ANTI-PIRACY)

### License Key Format
```
OFASTX-XXXX-XXXX-XXXX-XXXX-{ENCRYPTED_DOMAIN_HASH}
```

### Validation Process
```php
/**
 * License Validation (Daily Check)
 */
function ofastx_validate_license() {
    $license_key = get_option('ofastx_license_key');
    $domain = parse_url(home_url(), PHP_URL_HOST);
    
    // Build validation request
    $response = wp_remote_post('https://license.ofastshop.com/api/validate', [
        'body' => [
            'license_key' => $license_key,
            'domain' => $domain,
            'version' => OFASTX_VERSION,
            'php_version' => PHP_VERSION,
            'wp_version' => get_bloginfo('version'),
            'site_hash' => ofastx_generate_site_hash() // Unique site fingerprint
        ]
    ]);
    
    $data = json_decode(wp_remote_retrieve_body($response), true);
    
    if (!$data['valid']) {
        // License invalid - deactivate plugin
        ofastx_deactivate_plugin();
        add_action('admin_notices', 'ofastx_license_invalid_notice');
    }
    
    update_option('ofastx_last_license_check', time());
}

// Schedule daily check
add_action('ofastx_daily_license_check', 'ofastx_validate_license');
wp_schedule_event(time(), 'daily', 'ofastx_daily_license_check');
```

### Anti-Cracking Measures
1. **Domain Hash:** License tied to specific domain (cannot be transferred)
2. **Site Fingerprint:** Unique hash based on:
   - Database prefix
   - Site URL
   - WordPress install path
   - Admin email hash
3. **Daily Validation:** Background check (wp-cron)
4. **Code Obfuscation:** Critical license code obfuscated (ionCube/Zend Guard)
5. **Server-Side Validation:** Cannot be bypassed client-side
6. **Grace Period:** 7 days offline grace period (if validation fails)
7. **Deactivation:** Disables all modules if license invalid

### Emergency Deactivation
If license invalid:

```sql
-- 1. Email Logs
CREATE TABLE wp_ofast_email_logs (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    subject VARCHAR(255),
    sent_at DATETIME,
    recipient_count INT,
    status ENUM('sent', 'scheduled', 'failed') DEFAULT 'sent',
    notes TEXT,
    INDEX idx_sent_at (sent_at),
    INDEX idx_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 2. Newsletter Subscribers
CREATE TABLE wp_ofast_newsletter_subscribers (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    name VARCHAR(255),
    status ENUM('pending', 'confirmed', 'unsubscribed') DEFAULT 'pending',
    whatsapp_opted_in TINYINT(1) DEFAULT 0,
    whatsapp_number VARCHAR(20),
    subscribed_at DATETIME,
    confirmed_at DATETIME,
    unsubscribed_at DATETIME,
    ip_address VARCHAR(45),
    user_agent TEXT,
    INDEX idx_email (email),
    INDEX idx_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 3. Code Snippets
CREATE TABLE wp_ofast_snippets (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    description TEXT,
    code LONGTEXT NOT NULL,
    language ENUM('php', 'javascript', 'css') DEFAULT 'php',
    active TINYINT(1) DEFAULT 0,
    tags TEXT,
    scope ENUM('global', 'admin', 'frontend') DEFAULT 'global',
    priority INT DEFAULT 10,
    created_at DATETIME,
    updated_at DATETIME,
    created_by BIGINT,
    INDEX idx_active (active),
    INDEX idx_language (language),
    INDEX idx_scope (scope)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 4. Contact Forms
CREATE TABLE wp_ofast_forms (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    fields LONGTEXT NOT NULL, -- JSON array
    settings LONGTEXT, -- JSON settings
    notifications LONGTEXT, -- JSON notifications config
    active TINYINT(1) DEFAULT 1,
    created_at DATETIME,
    updated_at DATETIME,
    INDEX idx_active (active)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 5. Form Submissions
CREATE TABLE wp_ofast_form_submissions (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    form_id BIGINT NOT NULL,
    data LONGTEXT NOT NULL, -- JSON submission data
    ip_address VARCHAR(45),
    user_agent TEXT,
    referer TEXT,
    status ENUM('unread', 'read', 'spam', 'trash') DEFAULT 'unread',
    submitted_at DATETIME,
    read_at DATETIME,
    INDEX idx_form_id (form_id),
    INDEX idx_status (status),
    INDEX idx_submitted_at (submitted_at),
    FOREIGN KEY (form_id) REFERENCES wp_ofast_forms(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 6. Redirects
CREATE TABLE wp_ofast_redirects (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    source_url VARCHAR(500) NOT NULL,
    target_url VARCHAR(500) NOT NULL,
    type ENUM('301', '302', '307') DEFAULT '301',
    is_regex TINYINT(1) DEFAULT 0,
    hits INT DEFAULT 0,
    last_accessed DATETIME,
    active TINYINT(1) DEFAULT 1,
    created_at DATETIME,
    created_by BIGINT,
    INDEX idx_source (source_url(255)),
    INDEX idx_active (active),
    INDEX idx_hits (hits)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 7. Redirect Logs
CREATE TABLE wp_ofast_redirect_logs (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    redirect_id BIGINT NOT NULL,
    accessed_at DATETIME,
    ip_address VARCHAR(45),
    user_agent TEXT,
    referer TEXT,
    INDEX idx_redirect_id (redirect_id),
    INDEX idx_accessed_at (accessed_at),
    FOREIGN KEY (redirect_id) REFERENCES wp_ofast_redirects(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```
